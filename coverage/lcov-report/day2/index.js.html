const processArray = function processArray(input) {
  // Split by lines.
  inputArray = input.split('\n');
 
  // Create a 2D array.
  spreadsheetArray = [];
  inputArray.forEach((row) => {
    spreadsheetArray.push(row.split('\t'));
  });
 
  // Turn into numbers.
  spreadsheetArray.forEach((row) => {
    row.forEach((cell, i) => {
      row[i] = parseInt(cell);
    })
  });
 
  return spreadsheetArray;
};
